EdgePhone.ai Blog – Embed Snippet

Copy the code block below and paste it into any page where you want the blog to appear. The iframe will automatically resize to fit its content on all screen sizes. 

<!-- EdgePhone.ai Blog Embed -->
<iframe
  id="epai-blog-iframe"
  src="https://YOUR-USERNAME.github.io/YOUR-REPO-NAME/"
  title="EdgePhone.ai Blog – Edge AI Daily"
  loading="lazy"
  referrerpolicy="no-referrer-when-downgrade"
  style="width:100%;border:none;display:block;min-height:400px;"
></iframe>

<script>
(function () {
  'use strict';
  var iframe = document.getElementById('epai-blog-iframe');
  if (!iframe) return;

  /*
   * SECURITY NOTE
   * ─────────────
   * We check event.origin against the blog's exact origin before acting
   * on any postMessage, preventing malicious iframes from injecting
   * arbitrary resize commands into your page.
   *
   * ► CHANGE THIS to match your actual GitHub Pages URL.
   */
  var BLOG_ORIGIN = 'https://YOUR-USERNAME.github.io';

  function onMessage(event) {
    // Origin check – only process messages from our own blog
    if (event.origin !== BLOG_ORIGIN) return;

    var data = event.data;
    if (
      data &&
      typeof data === 'object' &&
      data.type === 'epai-resize' &&
      typeof data.height === 'number' &&
      data.height > 0
    ) {
      iframe.style.height = data.height + 'px';
    }
  }

  window.addEventListener('message', onMessage, false);
})();
</script>
